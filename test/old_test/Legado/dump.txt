const jsonData = JSON.parse(document.body.innerText)

const searchText = "Fórum de Interação";

// Função para pesquisar o texto específico dentro do JSON
function searchInJson(jsonData, searchText) {
    const results = [];

    // Itera sobre cada objeto dentro do array "results"
    for (const item of jsonData.results) {
        // Verifica se o texto está presente na propriedade "title" ou "description"
        if (item.title.includes(searchText) || item.description.includes(searchText)) {
            results.push(item);
        }
    }

    return results;
}

// Chamada da função de pesquisa
const searchResults = searchInJson(jsonData, searchText);
